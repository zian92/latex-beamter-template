\newcommand{\bgpicture}[2]{
	\usebackgroundtemplate{
		\tikz[overlay,remember picture] \node[opacity=#1, at=(current page.center)] {
			%		\hspace{-1.2cm}
			\includegraphics[height=\paperheight]{#2}};
	}
}

\newenvironment{pictureframe}[2][]{%
	\usebackgroundtemplate{%
		\includegraphics[height=\paperheight,width=\paperwidth]{#1}
	}
	\begin{frame}\frametitle{#2}
	}
	{
	\end{frame}
	\usebackgroundtemplate{}
}

\newenvironment{pictureframebreaks}[2][]{%
	\usebackgroundtemplate{
		\includegraphics[height=\paperheight,width=\paperwidth]{#1}
	}
	\begin{frame}[allowframebreaks]\frametitle{#2}
	}
	{
	\end{frame}
	\usebackgroundtemplate{}
}

\setbeamerfont{frametitle}{size=\large, series=\bf}
